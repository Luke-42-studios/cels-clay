**Design:** A sidebar + content layout that demonstrates:
- Nested containers (Column > Row > Column)
- Text with different styles
- Spacer for flexible spacing
- Box with background color
- Relative sizing (no hardcoded dimensions)

**Contents:**

```c
#ifndef CELS_CLAY_EXAMPLE_UI_H
#define CELS_CLAY_EXAMPLE_UI_H

#include <cels-clay/clay_primitives.h>
#include <cels-clay/clay_layout.h>

/*
 * Shared UI composition for the minimal dual-renderer example.
 * This file is included by both main_ncurses.c and main_sdl3.c.
 * The entity tree is identical -- only the renderer module differs.
 */

/* Color palette -- high contrast, visible on both dark terminals and windows */
#define UI_BG_DARK     (Clay_Color){30, 30, 40, 255}
#define UI_BG_SIDEBAR  (Clay_Color){40, 45, 60, 255}
#define UI_BG_CONTENT  (Clay_Color){35, 38, 50, 255}
#define UI_BG_CARD     (Clay_Color){50, 55, 75, 255}
#define UI_FG_TITLE    (Clay_Color){220, 220, 240, 255}
#define UI_FG_TEXT     (Clay_Color){180, 185, 200, 255}
#define UI_FG_DIM      (Clay_Color){120, 125, 140, 255}
#define UI_ACCENT      (Clay_Color){100, 140, 220, 255}

/*
 * SidebarItem -- a simple labeled item in the sidebar
 */
CEL_Compose(SidebarItem) {
    cel_has(ClayContainerConfig,
        .direction = CLAY_LEFT_TO_RIGHT,
        .padding = (Clay_Padding){.left = 8, .right = 8, .top = 4, .bottom = 4},
        .width = CLAY_SIZING_GROW(0),
        .height = CLAY_SIZING_FIT(0)
    );
}

/*
 * Sidebar -- vertical navigation panel
 */
CEL_Compose(Sidebar) {
    cel_has(ClayContainerConfig,
        .direction = CLAY_TOP_TO_BOTTOM,
        .width = CLAY_SIZING_PERCENT(0.25f),
        .height = CLAY_SIZING_GROW(0),
        .padding = (Clay_Padding){.left = 4, .right = 4, .top = 8, .bottom = 8},
        .gap = 2,
        .bg = UI_BG_SIDEBAR
    );
}

/*
 * ContentCard -- a card in the content area
 */
CEL_Compose(ContentCard) {
    cel_has(ClayContainerConfig,
        .direction = CLAY_TOP_TO_BOTTOM,
        .width = CLAY_SIZING_GROW(0),
        .height = CLAY_SIZING_FIT(0),
        .padding = (Clay_Padding){.left = 8, .right = 8, .top = 6, .bottom = 6},
        .gap = 4,
        .bg = UI_BG_CARD
    );
}

/*
 * ContentArea -- main content panel
 */
CEL_Compose(ContentArea) {
    cel_has(ClayContainerConfig,
        .direction = CLAY_TOP_TO_BOTTOM,
        .width = CLAY_SIZING_GROW(0),
        .height = CLAY_SIZING_GROW(0),
        .padding = (Clay_Padding){.left = 8, .right = 8, .top = 8, .bottom = 8},
        .gap = 8,
        .bg = UI_BG_CONTENT
    );
}

/*
 * AppLayout -- root layout composition
 * Builds the entity tree that both renderers display.
 */
CEL_Compose(AppLayout) {
    cel_has(ClayContainerConfig,
        .direction = CLAY_LEFT_TO_RIGHT,
        .width = CLAY_SIZING_GROW(0),
        .height = CLAY_SIZING_GROW(0),
        .bg = UI_BG_DARK
    );
}

/*
 * build_ui() -- constructs the complete UI entity tree.
 * Call inside a ClaySurface scope in the CEL_Build block.
 */
static inline void build_ui(void) {
    AppLayout() {
        Sidebar() {
            ClayText(.text = "MENU", .color = UI_ACCENT, .font_size = 16) {}
            ClaySpacer(.height = CLAY_SIZING_FIXED(8)) {}
            SidebarItem() {
                ClayText(.text = "Dashboard", .color = UI_FG_TITLE) {}
            }
            SidebarItem() {
                ClayText(.text = "Settings", .color = UI_FG_TEXT) {}
            }
            SidebarItem() {
                ClayText(.text = "About", .color = UI_FG_DIM) {}
            }
            ClaySpacer() {}
            ClayText(.text = "v0.6", .color = UI_FG_DIM, .font_size = 12) {}
        }
        ContentArea() {
            ClayText(.text = "Dashboard", .color = UI_FG_TITLE, .font_size = 20) {}
            ContentCard() {
                ClayText(.text = "Welcome", .color = UI_ACCENT, .font_size = 16) {}
                ClayText(.text = "This is a minimal cels-clay example rendering on both NCurses and SDL3 backends.", .color = UI_FG_TEXT) {}
            }
            ContentCard() {
                ClayRow(.gap = 8) {
                    ClayBox(.width = CLAY_SIZING_GROW(0), .height = CLAY_SIZING_FIXED(40), .bg = UI_ACCENT) {}
                    ClayBox(.width = CLAY_SIZING_GROW(0), .height = CLAY_SIZING_FIXED(40), .bg = UI_BG_SIDEBAR) {}
                    ClayBox(.width = CLAY_SIZING_GROW(0), .height = CLAY_SIZING_FIXED(40), .bg = UI_BG_CARD) {}
                }
            }
            ClaySpacer() {}
            ClayText(.text = "cels-clay v0.6 -- Entity-Based UI + Dual Renderers", .color = UI_FG_DIM, .font_size = 12) {}
        }
    }
}

#endif /* CELS_CLAY_EXAMPLE_UI_H */
```

Notes:
- `CEL_Compose` is the v0.4 macro for inline compositions (verify from cels.h -- it may be `CEL_Compose` or a different name). If the API uses `CEL_Composition` for the implementation side only and compositions need `CEL_Define_Composition` + `CEL_Composition`, use the static manual pattern instead (same as ClaySurface).
- All sizing uses CLAY_SIZING_GROW, CLAY_SIZING_PERCENT, CLAY_SIZING_FIXED, CLAY_SIZING_FIT -- no hardcoded pixel/cell values.
- Colors are high-contrast dark theme, visible on both terminals and windows.
- The `build_ui()` function is static inline so it can be included from both main files without linker conflicts.

```c
/*
 * cels-clay Minimal Example -- NCurses Backend
 *
 * Demonstrates the same entity tree from ui.h rendering in a terminal.
 * Run: ./cels-clay-example-ncurses
 */

#include <cels/cels.h>
#include <cels-ncurses/ncurses_module.h>   // or appropriate cels-ncurses header
#include <cels-clay/clay_engine.h>
#include <cels-clay/clay_ncurses_renderer.h>
#include <cels-clay/clay_primitives.h>
#include <cels-clay/clay_layout.h>
#include "ui.h"

int main(void) {
    cels_main {
        cels_scope {
            cels_register(NCurses, Clay_Engine, Clay_NCurses);

            NCursesWindow(.title = "cels-clay example", .fps = 30) {
                /* ClaySurface dimensions: terminal cells divided by aspect ratio.
                 * NCurses text measurement reports widths in cell columns, but
                 * Clay layout uses aspect-ratio-compensated units. Typical:
                 * width = COLS / 2.0 (aspect ratio), height = LINES. */
                ClaySurface(.width = 80, .height = 24) {
                    build_ui();
                }
            }
        }

        cels_session {
            /* Frame loop -- CELS drives the ECS pipeline.
             * ClaySurface dimensions should be reactive to terminal resize.
             * For this minimal example, use fixed dimensions. */
            while (cels_running()) {
                cels_step();
            }
        }
    }
    return 0;
}
```

Notes:
- The exact cels-ncurses include and NCursesWindow composition name need to be verified from the cels-ncurses module. Look at the cels-ncurses headers for the correct module name and window composition.
- ClaySurface dimensions: for terminal rendering, the width should be COLS/aspect_ratio and height should be LINES. For the minimal example, use fixed values that work on a standard 80x24 terminal. A production app would use reactive dimensions from the window resize signal.
- The NCurses frame loop uses `cels_running()` + `cels_step()` which drives the ECS pipeline including PreStore (layout) and OnRender (ncurses render).

**Contents:**

```c
/*
 * cels-clay Minimal Example -- SDL3 Backend
 *
 * Demonstrates the same entity tree from ui.h rendering in a graphical window.
 * Run: ./cels-clay-example-sdl3
 */

#include <cels/cels.h>
#include <cels_sdl3.h>
#include <cels-clay/clay_engine.h>
#include <cels-clay/clay_sdl3_renderer.h>
#include <cels-clay/clay_primitives.h>
#include <cels-clay/clay_layout.h>
#include "ui.h"

int main(void) {
    cels_main {
        cels_scope {
            /* Configure SDL3 renderer with a font for text rendering */
            Clay_SDL3_configure(&(ClaySDL3Config){
                .font_path = "assets/font.ttf",  /* Provide a TTF font */
                .font_size = 16
            });

            cels_register(SDL3_Engine, Clay_Engine, Clay_SDL3);

            SDL3Context(.video = true) {
                SDL3Window(.title = "cels-clay example", .width = 1280, .height = 720) {
                    /* ClaySurface uses pixel dimensions for SDL3.
                     * Match window size for 1:1 pixel mapping. */
                    ClaySurface(.width = 1280, .height = 720) {
                        build_ui();
                    }
                }
            }
        }

        cels_session {
            /* Event/frame loop -- CELS drives the ECS pipeline.
             * SDL3 events (window close, resize) are handled by cels-sdl3 systems.
             * For a production app, ClaySurface dimensions should react to resize. */
            while (cels_running()) {
                cels_step();
            }
        }
    }
    return 0;
}
```

Notes:
- The SDL3 example uses pixel dimensions (1280x720) for ClaySurface -- no aspect ratio compensation needed since SDL3 uses pixel-accurate coordinates.
- A font file path is required for text rendering. The example uses a relative path. In production, this would be an absolute path or resource-locator.
- SDL3Context and SDL3Window compositions come from cels-sdl3.
- The frame loop is identical to NCurses (cels_running + cels_step) -- the ECS pipeline drives both layout and rendering.

**Add at the end of CMakeLists.txt:**

```cmake
# ============================================================================
# Example applications (optional)
# ============================================================================
option(CELS_CLAY_BUILD_EXAMPLES "Build cels-clay example applications" OFF)

if(CELS_CLAY_BUILD_EXAMPLES)
    # NCurses example (requires cels-ncurses)
    if(TARGET cels-ncurses)
        add_executable(cels-clay-example-ncurses
            ${CMAKE_CURRENT_SOURCE_DIR}/examples/minimal/main_ncurses.c
        )
        target_link_libraries(cels-clay-example-ncurses PRIVATE
            cels-clay
        )
        target_include_directories(cels-clay-example-ncurses PRIVATE
            ${CMAKE_CURRENT_SOURCE_DIR}/examples/minimal
        )
    endif()

    # SDL3 example (requires cels-sdl3)
    if(TARGET cels-sdl3)
        add_executable(cels-clay-example-sdl3
            ${CMAKE_CURRENT_SOURCE_DIR}/examples/minimal/main_sdl3.c
        )
        target_link_libraries(cels-clay-example-sdl3 PRIVATE
            cels-clay
        )
        target_include_directories(cels-clay-example-sdl3 PRIVATE
            ${CMAKE_CURRENT_SOURCE_DIR}/examples/minimal
        )
    endif()
endif()
```

Notes:
- Both example targets link against cels-clay (INTERFACE library). The conditional renderer sources and backend links are already inherited from cels-clay's target_link_libraries.
- Each example is guarded by its respective backend target availability.
- The CELS_CLAY_BUILD_EXAMPLES option defaults to OFF so that library consumers aren't forced to build examples.
- include_directories for examples/minimal allows `#include "ui.h"` to resolve.
