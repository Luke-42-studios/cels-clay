<objective>
Rewrite the layout system to read entity property components instead of calling layout function pointers.

Purpose: This is the core paradigm shift from v0.5 to v0.6. Instead of `layout_fn(world, entity)` dispatching to developer-written functions with raw CLAY() macros, the layout walker reads ClayContainerConfig/ClayTextConfig/etc. components and generates CLAY() calls internally. Developers never touch Clay macros -- they compose entities with property components.

Output: Rewritten clay_walk_entity() with component-presence dispatch, cleaned clay_layout.h with ClayUI and CEL_Clay_Layout removed, preserved tree walk infrastructure and internal macros.
</objective>

**Replace the clay_walk_entity() function body:**

Current pattern (v0.5):
```c
if (layout && layout->layout_fn) {
    layout->layout_fn(world, entity);  // call developer's layout function
} else {
    clay_walk_children(world, entity);  // transparent passthrough
}
```

New pattern (v0.6):
```c
static void clay_walk_entity(ecs_world_t* world, ecs_entity_t entity) {
    ecs_entity_t prev_entity = g_layout_current_entity;
    g_layout_current_entity = entity;

    // Check component presence in priority order
    const ClayContainerConfig* container = (const ClayContainerConfig*)
        ecs_get_id(world, entity, ClayContainerConfig_id);
    const ClayTextConfig* text = (const ClayTextConfig*)
        ecs_get_id(world, entity, ClayTextConfig_id);
    const ClaySpacerConfig* spacer = (const ClaySpacerConfig*)
        ecs_get_id(world, entity, ClaySpacerConfig_id);
    const ClayImageConfig* image = (const ClayImageConfig*)
        ecs_get_id(world, entity, ClayImageConfig_id);

    if (container) {
        emit_container(world, entity, container);
    } else if (text) {
        emit_text(world, entity, text);
    } else if (spacer) {
        emit_spacer(world, entity, spacer);
    } else if (image) {
        emit_image(world, entity, image);
    } else {
        // No layout component -- transparent passthrough
        clay_walk_children(world, entity);
    }

    g_layout_current_entity = prev_entity;
}
```

**Implement emit functions:**

1. **emit_container(world, entity, config):**
   ```c
   static void emit_container(ecs_world_t* world, ecs_entity_t entity,
                               const ClayContainerConfig* config) {
       // Check for additive ClayBorderStyle
       const ClayBorderStyle* border_style = (const ClayBorderStyle*)
           ecs_get_id(world, entity, ClayBorderStyle_id);

       // Build CLAY() call with properties from config
       CLAY({
           .id = _cel_clay_auto_id(__COUNTER__),
           .layout = {
               .layoutDirection = config->direction,
               .childGap = config->gap,
               .padding = config->padding,
               .sizing = {
                   .width = config->width,
                   .height = config->height
               },
               .childAlignment = config->alignment
           },
           .backgroundColor = config->bg,
           .clip = config->clip ? CLAY_CLIP_ELEMENT : CLAY_CLIP_NONE,
           // Add border config if ClayBorderStyle is present
           .border = border_style ? (Clay_BorderConfig){
               .width = border_style->border.width,
               .color = border_style->border.color
           } : (Clay_BorderConfig){0},
           .cornerRadius = border_style ? border_style->corner_radius : (Clay_CornerRadius){0}
       }) {
           clay_walk_children(world, entity);
       }
   }
   ```
   Note: Verify the exact Clay API struct names. Clay_BorderConfig may be named differently. Check clay.h for the correct field names used in the CLAY() element config. The border field in Clay's element config is checked to ensure the right sub-struct is used.

   For default sizing: if config->width and config->height are zero-initialized (all fields zero), they may produce 0-width/0-height elements. The compositions in Phase 7 should set defaults. But as a safety net in the emit function, treat zero-initialized Clay_SizingAxis as CLAY_SIZING_GROW(0).

2. **emit_text(world, entity, config):**
   ```c
   static void emit_text(ecs_world_t* world, ecs_entity_t entity,
                          const ClayTextConfig* config) {
       (void)world; (void)entity;  // text is a leaf node
       if (!config->text) return;

       // Convert const char* to Clay_String using the frame arena
       int32_t len = (int32_t)strlen(config->text);
       Clay_String clay_str;
       if (len > 0) {
           clay_str = _cel_clay_frame_arena_string(config->text, len);
       } else {
           clay_str = (Clay_String){ .chars = "", .length = 0 };
       }

       CLAY_TEXT(clay_str, CLAY_TEXT_CONFIG({
           .textColor = config->color,
           .fontSize = config->font_size,
           .fontId = config->font_id,
           .letterSpacing = config->letter_spacing,
           .lineHeight = config->line_height,
           .wrapMode = config->wrap
       }));
   }
   ```

3. **emit_spacer(world, entity, config):**
   ```c
   static void emit_spacer(ecs_world_t* world, ecs_entity_t entity,
                             const ClaySpacerConfig* config) {
       (void)world; (void)entity;  // spacer is a leaf node
       CLAY({
           .id = _cel_clay_auto_id(__COUNTER__),
           .layout = {
               .sizing = {
                   .width = config->width,
                   .height = config->height
               }
           }
       }) {}
   }
   ```

4. **emit_image(world, entity, config):**
   ```c
   static void emit_image(ecs_world_t* world, ecs_entity_t entity,
                            const ClayImageConfig* config) {

       CLAY({
           .id = _cel_clay_auto_id(__COUNTER__),
           .layout = {
               .sizing = {
                   .width = config->width,
                   .height = config->height
               }
           },
           .image = {
               .imageData = config->source,
               .sourceDimensions = {
                   .width = config->source_width,
                   .height = config->source_height
               }
           },
           .backgroundColor = config->bg,
           .cornerRadius = config->corner_radius
       }) {}
   }
   ```
