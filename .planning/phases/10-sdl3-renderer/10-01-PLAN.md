```c
CEL_Module(Clay_SDL3, init) {
    // 1. Find the SDL3 window to create renderer from.
    //    Query for SDL3_WindowComponent entities.
    //    Get SDL_Window* from the first ready window.
    //    Create SDL_Renderer:
    //      g_renderer = SDL_CreateRenderer(window, NULL);
    //    If no window found yet, the renderer creation may need to be
    //    deferred. Simplest approach: create renderer lazily in the
    //    render callback on first frame. Store a flag.

    // 2. Load TTF font:
    //    int font_size = g_sdl3_config.font_size > 0 ? g_sdl3_config.font_size : 16;
    //    g_font = TTF_OpenFont(g_sdl3_config.font_path, font_size);
    //    Log warning if font fails to load.

    // 3. Register text measurement callback (pixel-based):
    //    Clay_SetMeasureTextFunction(clay_sdl3_measure_text, NULL);

    // 4. Register render system at OnRender phase:
    ClayRenderableData_register();
    cels_entity_t comp_ids[] = { ClayRenderableData_id };
    cels_system_declare("SDL3_ClayRenderable_ClayRenderableData",
                        CELS_Phase_OnRender, clay_sdl3_render, comp_ids, 1);
}
```

Note on lazy initialization: SDL_CreateRenderer needs an SDL_Window* which may not exist at module init time (depends on entity creation order). The render callback should check if g_renderer is NULL and attempt initialization on first call. This is the common pattern for renderer init that depends on window state.

**Lazy renderer init helper:**
```c
static bool ensure_renderer_initialized(void) {
    if (g_renderer) return true;

    // Query for SDL3_WindowComponent to find the window
    ecs_world_t* world = cels_get_world(cels_get_context());
    ecs_iter_t it = ecs_each_id(world, SDL3_WindowComponent_id);
    while (ecs_each_next(&it)) {
        for (int i = 0; i < it.count; i++) {
            const SDL3_WindowComponent* wc = (const SDL3_WindowComponent*)
                ecs_get_id(world, it.entities[i], SDL3_WindowComponent_id);
            if (wc && wc->window && wc->state == SDL3_WINDOW_READY) {
                g_renderer = SDL_CreateRenderer(wc->window, NULL);
                if (!g_renderer) {
                    SDL_Log("SDL_CreateRenderer failed: %s", SDL_GetError());
                    return false;
                }
                // Load font now that we have renderer context
                if (g_sdl3_config.font_path) {
                    int size = g_sdl3_config.font_size > 0 ? g_sdl3_config.font_size : 16;
                    g_font = TTF_OpenFont(g_sdl3_config.font_path, (float)size);
                    if (!g_font) {
                        SDL_Log("TTF_OpenFont failed: %s", SDL_GetError());
                    }
                }
                return true;
            }
        }
    }
    return false;
}
```

**Text measurement callback (pixel-based):**
```c
static Clay_Dimensions clay_sdl3_measure_text(
    Clay_StringSlice text,
    Clay_TextElementConfig* config,
    void* userData)
{
    (void)config; (void)userData;
    if (!g_font || text.length <= 0 || !text.chars) {
        return (Clay_Dimensions){0, 0};
    }

    // Null-terminate for TTF API
    char buf_stack[512];
    char* buf = buf_stack;
    if (text.length >= (int32_t)sizeof(buf_stack)) {
        buf = (char*)malloc((size_t)text.length + 1);
        if (!buf) return (Clay_Dimensions){0, 0};
    }
    memcpy(buf, text.chars, (size_t)text.length);
    buf[text.length] = '\0';

    // Use TTF_GetStringSize for pixel dimensions
    int w = 0, h = 0;
    TTF_GetStringSize(g_font, buf, (size_t)text.length, &w, &h);

    if (buf != buf_stack) free(buf);
    return (Clay_Dimensions){ .width = (float)w, .height = (float)h };
}
```

**Render callback:**
```c
static void clay_sdl3_render(CELS_Iter* it) {
    int count = cels_iter_count(it);
    ClayRenderableData* data = (ClayRenderableData*)cels_iter_column(
        it, ClayRenderableData_id, sizeof(ClayRenderableData));

    for (int i = 0; i < count; i++) {
        if (!data[i].dirty) continue;
        if (!ensure_renderer_initialized()) continue;

        Clay_RenderCommandArray cmds = data[i].render_commands;

        for (int32_t j = 0; j < cmds.length; j++) {
            Clay_RenderCommand* cmd = Clay_RenderCommandArray_Get(&cmds, j);
            Clay_BoundingBox bb = cmd->boundingBox;
            SDL_FRect rect = {
                .x = bb.x, .y = bb.y,
                .w = bb.width, .h = bb.height
            };

            switch (cmd->commandType) {
                case CLAY_RENDER_COMMAND_TYPE_RECTANGLE: {
                    Clay_Color c = cmd->renderData.rectangle.backgroundColor;
                    SDL_SetRenderDrawColor(g_renderer,
                        (uint8_t)c.r, (uint8_t)c.g, (uint8_t)c.b, (uint8_t)c.a);
                    SDL_RenderFillRect(g_renderer, &rect);
                    break;
                }
                case CLAY_RENDER_COMMAND_TYPE_TEXT: {
                    render_sdl3_text(cmd);
                    break;
                }
                case CLAY_RENDER_COMMAND_TYPE_BORDER: {
                    render_sdl3_border(cmd);
                    break;
                }
                case CLAY_RENDER_COMMAND_TYPE_SCISSOR_START: {
                    SDL_Rect clip = {
                        .x = (int)bb.x, .y = (int)bb.y,
                        .w = (int)bb.width, .h = (int)bb.height
                    };
                    SDL_SetRenderClipRect(g_renderer, &clip);
                    break;
                }
                case CLAY_RENDER_COMMAND_TYPE_SCISSOR_END: {
                    SDL_SetRenderClipRect(g_renderer, NULL);
                    break;
                }
                case CLAY_RENDER_COMMAND_TYPE_IMAGE: {
                    // Render SDL_Texture from imageData pointer
                    if (cmd->renderData.image.imageData) {
                        SDL_Texture* tex = (SDL_Texture*)cmd->renderData.image.imageData;
                        SDL_RenderTexture(g_renderer, tex, NULL, &rect);
                    }
                    break;
                }
                default:
                    break;
            }
        }
    }
}
```

**Text rendering helper:**
```c
static void render_sdl3_text(Clay_RenderCommand* cmd) {
    Clay_TextRenderData* td = &cmd->renderData.text;
    Clay_StringSlice text = td->stringContents;
    if (text.length <= 0 || !text.chars || !g_font) return;

    // Null-terminate
    char buf_stack[512];
    char* buf = buf_stack;
    if (text.length >= (int32_t)sizeof(buf_stack)) {
        buf = (char*)malloc((size_t)text.length + 1);
        if (!buf) return;
    }
    memcpy(buf, text.chars, (size_t)text.length);
    buf[text.length] = '\0';

    // Render text to surface -> texture
    Clay_Color c = td->textColor;
    SDL_Color sdl_color = {(uint8_t)c.r, (uint8_t)c.g, (uint8_t)c.b, (uint8_t)c.a};
    SDL_Surface* surface = TTF_RenderText_Blended(g_font, buf, (size_t)text.length, sdl_color);
    if (surface) {
        SDL_Texture* texture = SDL_CreateTextureFromSurface(g_renderer, surface);
        if (texture) {
            SDL_FRect dst = {
                .x = cmd->boundingBox.x,
                .y = cmd->boundingBox.y,
                .w = (float)surface->w,
                .h = (float)surface->h
            };
            SDL_RenderTexture(g_renderer, texture, NULL, &dst);
            SDL_DestroyTexture(texture);
        }
        SDL_DestroySurface(surface);
    }

    if (buf != buf_stack) free(buf);
}
```
Note: Verify SDL3_ttf API for TTF_RenderText_Blended. In SDL3_ttf it may be TTF_RenderText_Blended(font, text, length, color) or similar. Check actual function signature.

**Border rendering helper:**
```c
static void render_sdl3_border(Clay_RenderCommand* cmd) {
    Clay_BorderRenderData* bd = &cmd->renderData.border;
    Clay_BoundingBox bb = cmd->boundingBox;
    Clay_Color c = bd->color;
    SDL_SetRenderDrawColor(g_renderer,
        (uint8_t)c.r, (uint8_t)c.g, (uint8_t)c.b, (uint8_t)c.a);

    float x = bb.x, y = bb.y, w = bb.width, h = bb.height;

    // Draw each side as a line
    if (bd->width.top > 0) {
        SDL_RenderLine(g_renderer, x, y, x + w, y);
    }
    if (bd->width.bottom > 0) {
        SDL_RenderLine(g_renderer, x, y + h, x + w, y + h);
    }
    if (bd->width.left > 0) {
        SDL_RenderLine(g_renderer, x, y, x, y + h);
    }
    if (bd->width.right > 0) {
        SDL_RenderLine(g_renderer, x + w, y, x + w, y + h);
    }
}
```
